const defaultConfig = { button_label: 'Klik Di Sini', background_color: '#667eea', text_color: '#ffffff', button_color: '#ffffff', accent_color: '#764ba2' }; // Current data storage let currentData = []; let currentLinkUrl = 'https://google.com'; // DOM Elements const mainButton = document.getElementById('mainButton'); const buttonText = document.getElementById('buttonText'); const linkInput = document.getElementById('linkInput'); const saveBtn = document.getElementById('saveBtn'); const saveBtnIcon = document.getElementById('saveBtnIcon'); const saveBtnText = document.getElementById('saveBtnText'); const currentLinkDisplay = document.getElementById('currentLink'); const toast = document.getElementById('toast'); const toastMessage = document.getElementById('toastMessage'); // Show toast notification function showToast(message, isError = false) { toastMessage.textContent = message; toast.className = `fixed bottom-6 left-1/2 transform -translate-x-1/2 ${isError ? 'bg-red-500' : 'bg-green-500'} text-white px-6 py-3 rounded-full shadow-lg toast`; toast.classList.remove('hidden'); setTimeout(() => { toast.classList.add('hidden'); }, 3000); } // Update UI with current link function updateLinkDisplay(url) { mainButton.href = url; linkInput.value = url; currentLinkDisplay.textContent = url; currentLinkUrl = url; } // Set loading state function setLoading(isLoading) { saveBtn.disabled = isLoading; if (isLoading) { saveBtnIcon.innerHTML = ''; saveBtnIcon.setAttribute('class', 'w-5 h-5 animate-spin'); saveBtnText.textContent = 'Menyimpan...'; } else { saveBtnIcon.innerHTML = ''; saveBtnIcon.setAttribute('class', 'w-5 h-5'); saveBtnText.textContent = 'Simpan Link'; } } // Data SDK handler const dataHandler = { onDataChanged(data) { currentData = data; if (data.length > 0) { const latestLink = data[data.length - 1]; if (latestLink.link_url) { updateLinkDisplay(latestLink.link_url); } } } }; // Save link function async function saveLink() { const newUrl = linkInput.value.trim(); if (!newUrl) { showToast('Masukkan URL terlebih dahulu!', true); return; } setLoading(true); try { // If there's existing data, update it; otherwise create new if (currentData.length > 0) { const existingRecord = currentData[currentData.length - 1]; existingRecord.link_url = newUrl; const result = await window.dataSdk.update(existingRecord); if (result.isOk) { showToast('Link berhasil diperbarui!'); } else { showToast('Gagal menyimpan link. Coba lagi.', true); } } else { // Check data limit if (currentData.length >= 999) { showToast('Batas penyimpanan tercapai!', true); setLoading(false); return; } const result = await window.dataSdk.create({ link_url: newUrl, button_text: defaultConfig.button_label }); if (result.isOk) { showToast('Link berhasil disimpan!'); } else { showToast('Gagal menyimpan link. Coba lagi.', true); } } } catch (error) { showToast('Terjadi kesalahan. Coba lagi.', true); } setLoading(false); } // Element SDK render function async function onConfigChange(config) { const label = config.button_label || defaultConfig.button_label; buttonText.textContent = label; } // Initialize Element SDK if (window.elementSdk) { window.elementSdk.init({ defaultConfig, onConfigChange, mapToCapabilities: (config) => ({ recolorables: [ { get: () => config.background_color || defaultConfig.background_color, set: (value) => { config.background_color = value; window.elementSdk.setConfig({ background_color: value }); } }, { get: () => config.accent_color || defaultConfig.accent_color, set: (value) => { config.accent_color = value; window.elementSdk.setConfig({ accent_color: value }); } }, { get: () => config.text_color || defaultConfig.text_color, set: (value) => { config.text_color = value; window.elementSdk.setConfig({ text_color: value }); } } ], borderables: [], fontEditable: undefined, fontSizeable: undefined }), mapToEditPanelValues: (config) => new Map([ ['button_label', config.button_label || defaultConfig.button_label] ]) }); } // Initialize Data SDK async function initDataSdk() { if (window.dataSdk) { const result = await window.dataSdk.init(dataHandler); if (!result.isOk) { console.error('Failed to initialize Data SDK'); } } } // Event listeners saveBtn.addEventListener('click', saveLink); // Allow Enter key to save linkInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') { saveLink(); } }); // Initialize initDataSdk();
